from docx import Document

# Load passwords from the text file
def load_passwords(txt_file):
    with open(txt_file, "r", encoding="utf-8") as f:
        return [line.strip() for line in f.readlines()]

# Extract text from Word document
def extract_text_from_docx(docx_file):
    doc = Document(docx_file)
    return "\n".join([para.text for para in doc.paragraphs])

# Check if passwords exist in the Word file
def check_passwords(txt_file, docx_file):
    passwords = load_passwords(txt_file)
    doc_text = extract_text_from_docx(docx_file)
    
    found_passwords = [pwd for pwd in passwords if pwd in doc_text]
    
    if found_passwords:
        print("Passwords found in document:", found_passwords)
    else:
        print("No passwords found in the document.")

# Example usage
check_passwords("passwords.txt", "document.docx")
